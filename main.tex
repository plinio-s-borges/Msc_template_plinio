% QUICK START
% 1) Edit metadata in settings/authors.tex (title, author, dates, supervisors).
% 2) Add chapters in mainmatter/chapters/ and include them in sections/mainmatter.tex.
% 3) Add appendices in backmatter/appendix/ and include them in sections/backmatter.tex.
% 4) Add references to frontmatter/bibliography/references.bib and cite with \cite{key}.
% 5) Customize styles and packages in the settings/*.tex files.
% ----------------------------------------------------------------------
% STRUCTURE MAP
% - main.tex                  → This file, the document root.
% - settings/                 → Packages, layout, typography, commands, and metadata.
% - sections/frontmatter.tex  → Cover, title page, abstracts, ToC, lists of figures/tables.
% - sections/mainmatter.tex   → Main thesis chapters (numbered, Arabic page numbers).
% - sections/backmatter.tex   → Appendices and bibliography.
% ----------------------------------------------------------------------

% Preamble
\documentclass[a4paper, 11pt]{book}

%% ======================================================================
%% MODULAR SETTINGS
%% ======================================================================
\input{settings/packages}        % All package imports
\input{settings/layout}          % Page geometry and layout  
\input{settings/typography}      % Font settings and spacing, and typograhy, symbols
\input{settings/chapter_format}  % Chapter formatting
\input{settings/code_style}      % Code listing styles
\input{settings/authors} % Custom commands and macros (title, names, etc.)
\input{settings/page_style}      % Header and footer styling
\input{settings/comands}         % Custom commands and macros

%% Bibliography file
\addbibresource{frontmatter/bibliography/references.bib}

%% ======================================================================
%% DOCUMENT START
%% ======================================================================
\begin{document}

% Load the modular document structure 
\input{sections/frontmatter}
\input{sections/mainmatter}
\input{sections/backmatter}

\end{document}
